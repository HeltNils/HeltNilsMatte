  <script>
    function tilfeldigTall(min, maks) {
      return Math.floor(Math.random() * (maks - min + 1)) + min;
    }

    function settRuteord(id, tekst) {
      const el = document.getElementById(id);
      el.innerHTML = "";
      for (const tegn of tekst) {
        const span = document.createElement("span");
        span.innerHTML = tegn === " " ? "&nbsp;" : tegn;
        el.appendChild(span);
      }
    }

    function toSifre(n) {
      return String(n).padStart(2, " ");
    }

    function lagVertikalOppgave(tall1, operator, tall2) {
      const a = toSifre(tall1).split("");
      const b = toSifre(tall2).split("");

      return `
        <div class="vertical-problem">
          <div class="cell"></div>
          <div class="cell">${a[0] === " " ? "&nbsp;" : a[0]}</div>
          <div class="cell">${a[1]}</div>

          <div class="cell">${operator}</div>
          <div class="cell">${b[0] === " " ? "&nbsp;" : b[0]}</div>
          <div class="cell">${b[1]}</div>

          <div class="math-line"></div>

          <div class="cell">=</div>
          <div class="cell"></div>
          <div class="cell"></div>

          <div class="answer-line"></div>
          <div class="answer-line"></div>
        </div>
      `;
    }

    function lagHorisontalOppgave(tall1, operator, tall2, strekLengde = 3) {
      const tegn = `${tall1}${operator}${tall2}=`.split("");
      const cols = 11;
      let html = `<div class="horizontal-problem">`;

      for (let i = 0; i < cols; i++) {
        const tegnIndex = i;
        if (tegnIndex < tegn.length) {
          html += `<div class="cell">${tegn[tegnIndex]}</div>`;
        } else {
          html += `<div class="cell"></div>`;
        }
      }

      for (let i = 0; i < cols; i++) {
        if (i >= tegn.length && i < tegn.length + strekLengde) {
          html += `<div class="h-answer-line"></div>`;
        } else {
          html += `<div></div>`;
        }
      }

      for (let i = 0; i < cols; i++) {
        if (i >= tegn.length && i < tegn.length + strekLengde) {
          html += `<div class="h-answer-line"></div>`;
        } else {
          html += `<div></div>`;
        }
      }

      html += `</div>`;
      return html;
    }

    function lagAddisjon() {
      const a = tilfeldigTall(10, 99);
      const b = tilfeldigTall(10, 99);
      document.getElementById("addisjon").innerHTML = lagVertikalOppgave(a, "+", b);
    }

    function lagSubtraksjon() {
      const a = tilfeldigTall(20, 99);
      const b = tilfeldigTall(10, a);
      document.getElementById("subtraksjon").innerHTML = lagVertikalOppgave(a, "-", b);
    }

    function lagMultiplikasjon() {
      const a = tilfeldigTall(10, 99);
      const b = tilfeldigTall(10, 99);
      document.getElementById("multiplikasjon").innerHTML = lagHorisontalOppgave(a, "·", b, 3);
    }

    function lagDivisjon() {
      const divisor = tilfeldigTall(2, 9);
      const svar = tilfeldigTall(10, 99);
      const dividend = divisor * svar;
      document.getElementById("divisjon").innerHTML = lagHorisontalOppgave(dividend, ":", divisor, 3);
    }

    function lagOppgaver() {
      lagAddisjon();
      lagSubtraksjon();
      lagMultiplikasjon();
      lagDivisjon();
    }

    settRuteord("mainTitle", "Helt Nils lekser Nivå 1");
    settRuteord("titleAddisjon", "Addisjon");
    settRuteord("titleSubtraksjon", "Subtraksjon");
    settRuteord("titleMultiplikasjon", "Multiplikasjon");
    settRuteord("titleDivisjon", "Divisjon");
    settRuteord("symbolRow", "123456789+-·:=");

    lagOppgaver();
